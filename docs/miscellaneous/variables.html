Default value : (name: string, mandatory = true) => { let layer = undefined if (!layers) { throw new Error('Tiled Error: No layers found.') } for (let i = 0; i < layers.length; i++) { if (layers[i].name === name) { layer = layers[i] break } } if (!layer && mandatory) { throw new Error(`Tiled Error: Missing layer "${name}".`) } return layer }

Default value : (type: string, mandatory = false): Tiled.Object[] => { const es = [] for (let i = 0; i < entities.length; i++) { if (entities[i].type === type) { es.push(entities[i]) } } if (!es.length && mandatory) { throw new Error(`Tiled Error: Missing an object of type "${type}".`) } const converted = [] for (let i = 0; i < es.length; i++) { converted.push(convertEntity(es[i])) } return converted as any }
