export class HitBox {
  private _x: number
  private _y: number

  private _height: number
  private _width: number

  /** Gets the offset along the X axis from top left. */
  get x() {
    return this._x
  }

  /** Gets the offset along the Y axis from top left. */
  get y() {
    return this._y
  }

  /** Gets the width of the hitbox. */
  get width() {
    return this._width
  }

  /** Gets the height of the hitbox. */
  get height() {
    return  this._height
  }

  /**
   * Initialize HitBox object.
   *
   * @param x X axis offset from top left.
   * @param y Y axis offset from top left.
   * @param width Width of hitbox.
   * @param height Height of hitbox.
   */
  constructor(x: number, y: number, width: number, height: number) {
    this._x = x
    this._y = y

    this._height = height
    this._width = width
  }

  /**
   * Set new values for the hitbox.
   *
   * @param x X axis offset from top left.
   * @param y Y axis offset from top left.
   * @param width Width of hitbox.
   * @param height Height of hitbox.
   */
  set(x: number, y: number, width: number, height: number) {
    this._x = x
    this._y = y

    this._height = height
    this._width = width
  }
}
